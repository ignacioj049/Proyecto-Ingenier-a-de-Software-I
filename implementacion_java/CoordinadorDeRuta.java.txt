public class CoordinadorRuta extends Usuario {
    private String areaRuta;

    public CoordinadorRuta(int id, String nombre, String email, String telefono,
                           String area) {
        super(id, nombre, email, telefono);
        this.areaRuta = area;
    }

    public Ruta crearRuta(List<SolicitudRetiro> solicitudes) {
        return new Ruta(solicitudes);
    }
}
