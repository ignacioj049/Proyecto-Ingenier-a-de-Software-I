public class Cliente extends Usuario {
    private String direccionCliente;
    private String coordenadaCliente;

    public Cliente(int id, String nombre, String email, String telefono,
                   String direccion, String coordenada) {
        super(id, nombre, email, telefono);
        this.direccionCliente = direccion;
        this.coordenadaCliente = coordenada;
    }

    public SolicitudRetiro realizarSolicitud(String direccionRetiro) {
        return new SolicitudRetiro(direccionRetiro);
    }
}
